/**
 * @module routes/products.routes
 * @group Products
 */

import express from "express";

import { authenticateToken, checkAdmin } from "../middleware/index.js";
import {
  createProductHandler,
  deleteProductHandler,
  getProductByIdHandler,
  getProducts,
  updateProductHandler,
} from "../controllers/products.controller.js";

const router = express.Router();

/**
 * @route   GET /products
 * @group   Products
 * @desc    Fetch all published products
 * @access  Public
 * @returns {Promise<Array>} Array of published product documents
 */
router.get("/", getProducts);

/**
 * @route   GET /products/:id
 * @group   Products
 * @desc    Fetch a published product by ID
 * @access  Public
 * @param   {String} id - Product ID
 * @returns {Promise<Product>} Product document or null if not found
 */
router.get("/:id", getProductByIdHandler);

/**
 * @route   POST /products
 * @group   Products
 * @desc    Create a new product
 * @access  Admin (Requires authentication and admin role)
 * @param   {Object} req.body - Product data
 * @param   {String} req.body.name        - Name of the product
 * @param   {String} req.body.description - Description of the product
 * @param   {Number} req.body.price       - Product price in local currency (min: 0)
 * @param   {String[]} req.body.images    - Array of image URLs (max: 5)
 * @param   {String} req.body.category    - Category label (lowercase)
 * @param   {Number} req.body.stock       - Units in stock (min: 0)
 * @returns {Promise<Product>} Created product document
 * @example
 * curl -X POST \
 *   http://localhost:3000/products \
 *   -H 'Content-Type: application/json' \
 *   -d '{"name":"Product Name","description":"Product description","price":1.99,"images":["https://example.com/image.jpg"],"category":"category","stock":5}'
 */
router.post("/", authenticateToken, checkAdmin, createProductHandler);

/**
 * @route   PUT /products/:id
 * @group   Products
 * @desc    Update product details
 * @access  Admin (Requires authentication and admin role)
 * @param   {String} id - Product ID
 * @param   {Object} req.body - Product data
 * @param   {String} req.body.name        - Name of the product
 * @param   {String} req.body.description - Description of the product
 * @param   {Number} req.body.price       - Product price in local currency (min: 0)
 * @param   {String[]} req.body.images    - Array of image URLs (max: 5)
 * @param   {String} req.body.category    - Category label (lowercase)
 * @param   {Number} req.body.stock       - Units in stock (min: 0)
 * @returns {Promise<Product>} Updated product document
 * @example
 * curl -X PUT \
 *   http://localhost:3000/products/:id \
 *   -H 'Content-Type: application/json' \
 *   -d '{"name":"Product Name","description":"Product description","price":1.99,"images":["https://example.com/image.jpg"],"category":"category","stock":5}'
 */
router.put("/:id", authenticateToken, checkAdmin, updateProductHandler);

/**
 * @route   DELETE /products/:id
 * @group   Products
 * @desc    Delete a product by ID
 * @access  Admin (Requires authentication and admin role)
 * @param   {String} id - Product ID
 * @returns {Promise<void>}
 */
router.delete("/:id", authenticateToken, checkAdmin, deleteProductHandler);

export default router;
